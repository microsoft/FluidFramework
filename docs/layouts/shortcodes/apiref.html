{{/* Helper template for linking to an API member's generated docs. */}} {{/* Parameters: */}} {{/* - name: The API member name. Note that this must be an item for which a document is rendered. */}} {{/* This includes classes, interfaces, and namespaces. */}} {{/* - kind: The kind of API item (must be "class", "interface", or "namespace"). */}} {{/* - (optional) unscopedPackageName: The *unscoped* name of the package to which the API member belongs. */}} {{/* default value: "fluid-framework" */}} {{- $name := .Get 0 -}} {{- $kind := .Get 1 -}} {{- $unscopedPackageName := default "fluid-framework" (.Get 2) -}} {{- $fileNameWithoutExtension := lower (delimit (slice $name $kind) "-") -}} {{- $fileName := delimit (slice $fileNameWithoutExtension ".md") "" -}} {{- $path := delimit (slice "/docs/apis/" $unscopedPackageName "/" $fileName) "" -}} {{ $name }} {{- /* Strip trailing newline. */ -}}
