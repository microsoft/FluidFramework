{{/* Helper template for linking to the root page of a package's API docs. */}} {{/* Parameters: */}} {{/* - unscopedPackageName: The *unscoped* name of the package being linked to. */}} {{- $currentVersion := $.Site.Data.versions.currentVersion -}} {{- $apiLink := print "/docs/api/" $currentVersion "/" -}} {{- $unscopedPackageName := .Get 0 -}} {{- $path := delimit (slice $apiLink $unscopedPackageName ".md" ) "" -}} {{ $unscopedPackageName }} {{- /* Strip trailing newline. */ -}}
