{{/* Helper template for linking to the root page of a package's API docs. */}} {{/* Parameters: */}} {{/* - unscopedPackageName: The *unscoped* name of the package being linked to. */}} {{/* - (optional) version: The specific API docs version to point to. Default is the site's "current version". E.g. "v1" or "v2". */}} {{- $currentVersion := $.Site.Data.versions.currentVersion -}} {{- $apiLink := print "/docs/api/" $currentVersion "/" -}} {{- $unscopedPackageName := .Get 0 -}} {{- $version := default $.Site.Data.versions.params.currentVersion (.Get 1) -}} {{- $path := delimit (slice "/docs/api/" $version "/" $unscopedPackageName ".md" ) "" -}} {{ relref . $path }} {{- /* Strip trailing newline. */ -}}
